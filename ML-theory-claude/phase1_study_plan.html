import numpy as np # Vectors a = np.array([3, 4, 0]) b = np.array([1, 2, 2]) # Dot product — manually and with numpy dot_manual = sum(a[i]*b[i] for i in range(len(a))) dot_numpy = np.dot(a, b) print(f"Dot product: {dot_manual} = {dot_numpy}") # L2 norm (magnitude) norm_a = np.sqrt(np.sum(a**2)) print(f"||a|| = {norm_a:.4f}") # same as np.linalg.norm(a) # Cosine similarity (angle between vectors) cos_sim = np.dot(a, b) / (np.linalg.norm(a) * np.linalg.norm(b)) print(f"Cosine similarity: {cos_sim:.4f}") # This is exactly what vector search in RAG computes! # Real data: feature vector from breast cancer dataset from sklearn.datasets import load_breast_cancer cancer = load_breast_cancer(as_frame=True) row_1 = cancer.data.iloc[0].values # first patient = vector in 30D space row_2 = cancer.data.iloc[1].values # second patient dist = np.linalg.norm(row_1 - row_2) print(f"Distance between patients 1 and 2: {dist:.4f}")

import numpy as np # Matrix operations A = np.array([[1,2],[3,4],[5,6]]) # 3×2 matrix B = np.array([[7,8,9],[10,11,12]]) # 2×3 matrix AB = A @ B # matrix multiplication (3×2)×(2×3) = 3×3 print(f"A shape: {A.shape}, B shape: {B.shape}, AB shape: {AB.shape}") print(f"AB =\n{AB}") # Transpose print(f"Aᵀ =\n{A.T}") # Normal equation for linear regression: w = (XᵀX)⁻¹Xᵀy from sklearn.datasets import fetch_california_housing data = fetch_california_housing() X = np.hstack([data.data, np.ones((data.data.shape[0],1))]) # add bias y = data.target # Solve analytically (closed form) w = np.linalg.solve(X.T @ X, X.T @ y) print(f"\nLinear regression weights: {w.round(4)}") y_pred = X @ w ss_res = np.sum((y - y_pred)**2) ss_tot = np.sum((y - y.mean())**2) print(f"R² = {1 - ss_res/ss_tot:.4f}")

import numpy as np import matplotlib.pyplot as plt # Numerical derivative (finite difference) def f(x): return x**2 + 3*x + 2 def df(x): return 2*x + 3 # analytical derivative x = np.linspace(-5, 5, 100) h = 1e-5 numerical_deriv = (f(x+h) - f(x)) / h analytical_deriv = df(x) print("Max error between numerical and analytical:", np.max(np.abs(numerical_deriv - analytical_deriv))) # Chain rule example: MSE loss with linear prediction def mse(w, X, y): y_pred = X @ w return np.mean((y - y_pred)**2) def mse_gradient(w, X, y): y_pred = X @ w return -2/len(y) * X.T @ (y - y_pred) # chain rule! # One step of gradient descent np.random.seed(42) X_demo = np.random.randn(100, 3) y_demo = X_demo @ np.array([2, -1, 3]) + np.random.randn(100)*0.1 w = np.zeros(3) for step in range(50): grad = mse_gradient(w, X_demo, y_demo) w = w - 0.01 * grad # w = w - lr * gradient if step % 10 == 0: print(f" Step {step:3d}: loss={mse(w,X_demo,y_demo):.6f} w={w.round(3)}")

import numpy as np from scipy import stats # Bayes theorem — spam filter analogy P_spam = 0.30 # 30% of emails are spam P_word_spam = 0.80 # word appears in 80% of spam P_word_ham = 0.10 # word appears in 10% of non-spam P_word = P_word_spam*P_spam + P_word_ham*(1-P_spam) P_spam_word = P_word_spam * P_spam / P_word print(f"P(spam | contains word) = {P_spam_word:.4f}") # This is exactly what Naive Bayes does! Let's verify from sklearn.naive_bayes import BernoulliNB from sklearn.datasets import fetch_20newsgroups from sklearn.feature_extraction.text import CountVectorizer cats = ['alt.atheism','soc.religion.christian'] train = fetch_20newsgroups(subset='train',categories=cats) test = fetch_20newsgroups(subset='test', categories=cats) vec = CountVectorizer(binary=True, max_features=1000) nb = BernoulliNB() nb.fit(vec.fit_transform(train.data), train.target) print(f"\nNaive Bayes accuracy: {nb.score(vec.transform(test.data),test.target):.4f}") print("Naive Bayes IS repeated Bayes theorem per feature — same math!")

import numpy as np from scipy.optimize import minimize_scalar from scipy.stats import norm # MLE for Gaussian mean (should recover the true mean) np.random.seed(42) data = np.random.normal(5.0, 2.0, 200) # true mean=5, std=2 # Negative log-likelihood for Gaussian def neg_log_likelihood(mu): return -np.sum(norm.logpdf(data, loc=mu, scale=2.0)) result = minimize_scalar(neg_log_likelihood, bounds=(0,10), method='bounded') print(f"MLE estimate of mean: {result.x:.4f} (sample mean: {data.mean():.4f})") print("They match! MLE finds the sample mean for Gaussian distribution.") # MLE for logistic regression = cross-entropy loss def cross_entropy(y_true, y_pred): eps = 1e-9 return -np.mean(y_true*np.log(y_pred+eps) + (1-y_true)*np.log(1-y_pred+eps)) y_true = np.array([1,1,0,0,1]) for p in [0.9, 0.7, 0.5, 0.3]: y_pred = np.full(5, p) print(f" predicted={p}: BCE = {cross_entropy(y_true,y_pred):.4f}") print("Lower BCE = predictions more likely under Bernoulli MLE!")

If you fall behind — triage guide
Week behind — 1 day catchup plan
Skip the theory — go straight to code. Read code comments. Understand by running and modifying. Theory can be caught up on weekends. Code practice cannot be skipped.

Prioritise: gradient descent > bias-variance > loss functions > everything else
Struggling with math (Week 1-2)
Don't get blocked on math rigor. Accept the formula, understand what it computes, implement it in numpy, see the result. Deep mathematical proof can wait until you have practical understanding. Use Khan Academy for specific gaps.
Ahead of schedule
Start Kaggle competitions early. Enter Titanic, Housing Prices, and Heart Disease. Read top public notebooks. Try to beat the current score. This is the fastest way to go deeper than the study plan covers.
The single best indicator you are on track: At the end of each week, open a blank Python file and implement that week's core algorithm from scratch in 20 minutes. Week 1 = gradient descent. Week 3 = logistic regression. Week 5 = decision tree. Week 6 = XGBoost forward pass (simplified). If you can do this without notes, you truly understand it. If you cannot, you memorised code without understanding it — that does not survive an interview or production problem.
